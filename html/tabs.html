import { Tabs } from '../js/modules/tabs.mjs'
    new Tabs('#demo', {
        trigger: 'click',
        swipeable: true,
        animation: 'slide',
        duration: 500,
        itemCLS: 'tabs-item',
        activeCLS: 'tabs-active',
        navCLS: 'tabs-nav',
        contentCLS: 'tabs-content',
        trackCLS: 'tabs-track',
        panelCLS: 'tabs-panel'
    })
参数
参数名 类型 默认值 取值范围/示例 功能说明
trigger string 'click' 'click' │ 'hover' 切换方式：点击 或 鼠标悬停（浮动）
swipeable boolean true true │ false 移动端是否允许左右滑动切换
swipeable boolean true true │ false 移动端是否开启“左右滑动切换”手势
lazy boolean false true │ false true 时仅在首次切换到某面板才执行加载（可自定义 loadPanel 逻辑）
animation string 'slide-fade' 'slide' │ 'fade' │ 'slide-fade' │ 'none' 面板切换动画效果，共 4 种可选
duration number 300 ≥ 0（单位 ms） 动画持续时长，0 等价于无动画
threshold number 50 20 ~ 100（单位 px） 左右滑动切换的“最小滑动距离”阈值，低于此值视为误触
bound boolean true true │ false true 到头/尾后循环切换；false 到边界后停止
itemCLS string 'tabs-item' 自定义 CSS 类名 单个标题按钮的 class（用来绑定点击事件）
activeCLS string 'tabs-active' 自定义 CSS 类名 当前激活标题按钮的高亮 class
navCLS string 'tabs-nav' 自定义 CSS 类名 标题栏容器 class
contentCLS string 'tabs-content' 自定义 CSS 类名 面板外包裹层 class（用于 overflow:hidden）
trackCLS string 'tabs-track' 自定义 CSS 类名 面板轨道 flex 容器，滑动动画时 translateX 作用对象
panelCLS string 'tabs-panel' 自定义 CSS 类名 单个面板区块 class
