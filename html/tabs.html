                                            //普通文档切换（无花哨动画，最稳）
                                            import { Tabs } from './tabs.mjs';
                                            const myTabs = new Tabs('#myTabs', {
                                                // 基础
                                                mode: 'slide',          // 模式: slide, fade, static
                                                defaultIndex: 0,        // 默认第几个
                                                duration: 300,          // 动画时间
                                                
                                                // 高级
                                                autoHeight: true,       // 开启高度自适应 (推荐)
                                                hashSync: true,         // URL 变为 #tab-id，刷新后保持状态
                                                keyboard: true,         // 支持键盘左右键切换
                                                
                                                // 移动端
                                                swipeable: true,        // 开启触屏滑动
                                                swipeThreshold: 50,     // 滑动多少像素切换
                                                velocityThreshold: 0.3, // 快速轻扫的灵敏度
                                                
                                                // 自动播放 (轮播模式)
                                                autoplay: false,        
                                                interval: 3000,
                                                pauseOnHover: true,
                                                
                                                // 回调
                                                onChange: (index, panel) => {
                                                    console.log('Switched to:', index);
                                                    // 这里可以做懒加载，比如:
                                                    // const img = panel.querySelector('img[data-src]');
                                                    // if(img) img.src = img.dataset.src;
                                                }
                                            });
                                            // 手动控制
                                            // myTabs.next();
                                            // myTabs.activate(2);
                                            // myTabs.destroy();
参数
参数名 类型 默认值 取值范围/示例 功能说明
trigger string 'click' 'click' │ 'hover' 切换方式：点击 或 鼠标悬停（浮动）
swipeable boolean true true │ false （移动端）是否允许左右滑动切换
swipeable boolean true true │ false （移动端）是否开启“左右滑动切换”手势
animation string 'static' 'slide' │ 'fade' │ 'static' 面板切换动画效果，共 3 种可选
duration number 300 ≥ 0（单位 ms） 动画持续时长，0 等价于无动画
swipeThreshold number 40 20 ~ 100（单位 px） （移动端）左右滑动切换的“最小滑动距离”阈值，低于此值视为误触
velocityThreshold number 0.3 0 ~ 1（单位 px/ms） （移动端）滑动速度阈值，低于此值视为静态点击
hashSync boolean false true │ false （高级功能）是否开启 URL 哈希同步
keyboard boolean true true │ false （高级功能）是否开启键盘导航
autoplay boolean false true │ false （高级功能）是否开启自动播放
interval number 3000 ≥ 0（单位 ms） （高级功能）自动播放间隔，0 等价于无自动播放
pauseOnHover boolean true true │ false （高级功能）是否在悬停时暂停自动播放
